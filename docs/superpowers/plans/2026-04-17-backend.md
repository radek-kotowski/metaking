# MetaKing Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Node.js/TypeScript/PostgreSQL backend that powers all asynchronous multiplayer mechanics: auth, player state, portal traversal, bot simulation, and leaderboard.

**Architecture:** Single Express app with REST endpoints, MikroORM for Postgres, node-cron for bot simulation. All game constants live in one file and are returned to the client on session start. No WebSockets — all interactions are discrete REST calls.

**Tech Stack:** Node.js 20, TypeScript 5, Express 5, MikroORM 6, PostgreSQL 16, node-cron, Jest + Supertest for tests, Vitest optional.

```typescript
import { CONSTANTS } from '../config/constants';

export interface WeaponResult {
  name: string;
  tier: number;
}

export function getWeaponTier(mana: number): (typeof CONSTANTS.WEAPON_TIERS)[number] {
  const tiers = [...CONSTANTS.WEAPON_TIERS].reverse();
  const match = tiers.find(t => mana >= t.minMana);
  return match ?? CONSTANTS.WEAPON_TIERS[0];
}

export function rollWeapon(mana: number): WeaponResult {
  const tierDef = getWeaponTier(mana);
  const weapons = tierDef.weapons as readonly string[];
  const name = weapons[Math.floor(Math.random() * weapons.length)];
  return { name, tier: tierDef.tier };
}
```

```typescript
import jwt from 'jsonwebtoken';
import { CONSTANTS } from '../config/constants';

export interface JwtPayload {
  playerId: string;
}

export function issueJwt(playerId: string, expiryDays?: number): string {
  const days = expiryDays ?? CONSTANTS.JWT_EXPIRY_DAYS;
  return jwt.sign({ playerId }, process.env.JWT_SECRET!, {
    expiresIn: `${days}d`,
  });
}

export function verifyJwt(token: string): JwtPayload {
  const payload = jwt.verify(token, process.env.JWT_SECRET!);
  if (typeof payload === 'string' || !('playerId' in payload)) {
    throw new Error('Invalid JWT payload');
  }
  return payload as JwtPayload;
}
```

```typescript
import { Request, Response, NextFunction } from 'express';
import { verifyJwt } from '../services/authService';

export interface AuthRequest extends Request {
  playerId?: string;
}

export function requireAuth(req: AuthRequest, res: Response, next: NextFunction): void {
  const header = req.headers.authorization;
  if (!header?.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Missing token' });
    return;
  }
  try {
    const payload = verifyJwt(header.slice(7));
    req.playerId = payload.playerId;
    next();
  } catch {
    res.status(401).json({ error: 'Invalid or expired token' });
  }
}
```

```typescript
import { Router } from 'express';
import { requireAuth, AuthRequest } from '../middleware/auth';
import { getEm } from '../db';
import { Player } from '../entities/Player';
import { Message } from '../entities/Message';
import { CONSTANTS } from '../config/constants';

const router = Router();

router.get('/me', requireAuth, async (req: AuthRequest, res, next) => {
  try {
    const em = getEm();
    const player = await em.findOneOrFail(Player, req.playerId!);

    const messages = await em.find(Message, { toPlayerId: player.id, seen: false }, {
      orderBy: { createdAt: 'ASC' },
    });

    // Mark messages seen
    for (const m of messages) m.seen = true;
    await em.flush();

    const totalManaStolen = messages.reduce((sum, m) => sum + m.manaStolen, 0);

    res.json({
      player: {
        id: player.id,
        nickname: player.nickname,
        mana: player.mana,
        stamina: player.stamina,
        hearts: player.hearts,
        weaponName: player.weaponName,
        weaponTier: player.weaponTier,
        isOnline: player.isOnline,
      },
      messages: messages.map(m => ({
        fromNickname: m.fromNickname,
        emoji: m.emoji,
        manaStolen: m.manaStolen,
        createdAt: m.createdAt,
      })),
      totalManaStolen,
      constants: CONSTANTS,
    });
  } catch (err) {
    next(err);
  }
});

router.post('/nickname', requireAuth, async (req: AuthRequest, res, next) => {
  try {
    const { nickname } = req.body;

    if (
      typeof nickname !== 'string' ||
      nickname.length < CONSTANTS.NICKNAME_MIN_LENGTH ||
      nickname.length > CONSTANTS.NICKNAME_MAX_LENGTH ||
      !/^[a-zA-Z0-9_]+$/.test(nickname)
    ) {
      res.status(400).json({ error: `Nickname must be ${CONSTANTS.NICKNAME_MIN_LENGTH}–${CONSTANTS.NICKNAME_MAX_LENGTH} alphanumeric chars or underscores` });
      return;
    }

    const em = getEm();
    const existing = await em.findOne(Player, { nickname });
    if (existing) {
      res.status(409).json({ error: 'Nickname already taken' });
      return;
    }

    const player = await em.findOneOrFail(Player, req.playerId!);
    player.nickname = nickname;
    await em.flush();

    res.json({ nickname: player.nickname });
  } catch (err) {
    next(err);
  }
});
```

```typescript
import { EntityManager } from '@mikro-orm/postgresql';
import { Portal } from '../entities/Portal';
import { Player } from '../entities/Player';
import { Bot } from '../entities/Bot';
import { Message } from '../entities/Message';
import { CONSTANTS } from '../config/constants';
import { rollWeapon } from './weaponService';

export interface PortalInfo {
  portalId: string;
  ownerNickname: string;
  ownerMana: number;
  isMetaKing: boolean;
}

export interface PortalEntryResult {
  success: boolean;
  error?: string;
  playerMana?: number;
  playerStamina?: number;
  newPortal?: PortalInfo;
}

export async function assignPortalsToPlayer(
  playerId: string,
  em: EntityManager,
): Promise<Portal[]> {
  // Deactivate any existing portals assigned to this player
  const existing = await em.find(Portal, { assignedToPlayerId: playerId, isActive: true });
  for (const p of existing) {
    p.isActive = false;
    p.assignedToPlayerId = null;
  }
  await em.flush();

  const portals: Portal[] = [];
  for (let i = 0; i < CONSTANTS.PORTAL_COUNT; i++) {
    const portal = await assignOnePortal(playerId, em);
    portals.push(portal);
  }
  return portals;
}

async function assignOnePortal(playerId: string, em: EntityManager): Promise<Portal> {
  // Try offline real players first (excluding self and already-assigned owners)
  const assigned = await em.find(Portal, { assignedToPlayerId: playerId, isActive: true });
  const excludedOwnerIds = assigned.map(p => p.ownerId).concat([playerId]);

  const offlinePlayer = await em.findOne(
    Player,
    { isOnline: false, id: { $nin: excludedOwnerIds } },
    { orderBy: { lastSeenAt: 'ASC' } },
  );

  if (offlinePlayer) {
    const portal = em.create(Portal, {
      ownerId: offlinePlayer.id,
      ownerType: 'player',
      isActive: true,
      assignedToPlayerId: playerId,
      assignedAt: new Date(),
    });
    await em.persistAndFlush(portal);
    return portal;
  }

  // Fall back to offline bot
  const offlineBot = await em.findOne(
    Bot,
    { isOnline: false, id: { $nin: excludedOwnerIds } },
    { orderBy: { mana: 'DESC' } },
  );

  if (!offlineBot) throw new Error('No offline targets available for portal assignment');

  const portal = em.create(Portal, {
    ownerId: offlineBot.id,
    ownerType: 'bot',
    isActive: true,
    assignedToPlayerId: playerId,
    assignedAt: new Date(),
  });
  await em.persistAndFlush(portal);
  return portal;
}

export async function executePortalEntry(
  playerId: string,
  portalId: string,
  emoji: string,
  em: EntityManager,
): Promise<PortalEntryResult> {
  const player = await em.findOneOrFail(Player, playerId);

  if (player.stamina < CONSTANTS.PORTAL_STAMINA_REQUIRED) {
    return { success: false, error: `Need ${CONSTANTS.PORTAL_STAMINA_REQUIRED} stamina to use a portal` };
  }

  const portal = await em.findOneOrFail(Portal, portalId);
  if (!portal.isActive || portal.assignedToPlayerId !== playerId) {
    return { success: false, error: 'Portal is no longer valid' };
  }

  // Race condition check: if target is a player, verify still offline
  if (portal.ownerType === 'player') {
    const target = await em.findOneOrFail(Player, portal.ownerId);
    if (target.isOnline) {
      portal.isActive = false;
      await em.flush();
      return { success: false, error: 'That player just came online — portal closed' };
    }
    // Deduct mana from target player
    target.mana = Math.max(CONSTANTS.MANA_MIN, target.mana - CONSTANTS.PORTAL_MANA_STEAL);
    // Record message
    const message = em.create(Message, {
      fromId: player.id,
      fromNickname: player.nickname ?? 'Unknown',
      toPlayerId: target.id,
      emoji,
      manaStolen: CONSTANTS.PORTAL_MANA_STEAL,
    });
    em.persist(message);
  } else {
    // Bot target
    const bot = await em.findOneOrFail(Bot, portal.ownerId);
    bot.mana = Math.max(CONSTANTS.MANA_MIN, bot.mana - CONSTANTS.PORTAL_MANA_STEAL);
  }

  // Update player
  player.mana += CONSTANTS.PORTAL_MANA_GAIN;
  player.stamina = Math.max(CONSTANTS.STAMINA_MIN, player.stamina - CONSTANTS.PORTAL_STAMINA_COST);

  // Check weapon milestone (crossed a tier boundary?)
  const weapon = rollWeapon(player.mana);
  if (weapon.tier > player.weaponTier) {
    player.weaponName = weapon.name;
    player.weaponTier = weapon.tier;
  }

  // Close portal, assign a replacement
  portal.isActive = false;
  portal.assignedToPlayerId = null;
  await em.flush();

  // Assign replacement portal
  const newPortal = await assignOnePortal(playerId, em);
  const newPortalInfo = await buildPortalInfo(newPortal, em);

  return {
    success: true,
    playerMana: player.mana,
    playerStamina: player.stamina,
    newPortal: newPortalInfo,
  };
}

export async function buildPortalInfo(portal: Portal, em: EntityManager): Promise<PortalInfo> {
  if (portal.ownerType === 'player') {
    const owner = await em.findOneOrFail(Player, portal.ownerId);
    // Determine if MetaKing = rank 1 on leaderboard (simplified: highest mana)
    const topPlayer = await em.findOne(Player, {}, { orderBy: { mana: 'DESC' } });
    return {
      portalId: portal.id,
      ownerNickname: owner.nickname ?? 'Unknown',
      ownerMana: owner.mana,
      isMetaKing: topPlayer?.id === owner.id,
    };
  } else {
    const bot = await em.findOneOrFail(Bot, portal.ownerId);
    const topBot = await em.findOne(Bot, {}, { orderBy: { mana: 'DESC' } });
    return {
      portalId: portal.id,
      ownerNickname: bot.nickname,
      ownerMana: bot.mana,
      isMetaKing: topBot?.id === bot.id,
    };
  }
}
```

```typescript
import { EntityManager } from '@mikro-orm/postgresql';
import { Player } from '../entities/Player';
import { Bot } from '../entities/Bot';
import { CONSTANTS } from '../config/constants';

export interface LeaderboardEntry {
  rank: number;
  nickname: string;
  mana: number;
  weaponTier: number;
  isMetaKing: boolean;
  type: 'player' | 'bot';
}

let cache: { entries: LeaderboardEntry[]; expiresAt: number } | null = null;

export async function getLeaderboard(em: EntityManager): Promise<LeaderboardEntry[]> {
  if (cache && Date.now() < cache.expiresAt) return cache.entries;

  const players = await em.find(Player, { nickname: { $ne: null } }, {
    orderBy: { mana: 'DESC' },
    limit: CONSTANTS.LEADERBOARD_TOP_N,
    fields: ['id', 'nickname', 'mana', 'weaponTier'],
  });

  const bots = await em.find(Bot, {}, {
    orderBy: { mana: 'DESC' },
    limit: CONSTANTS.LEADERBOARD_TOP_N,
    fields: ['id', 'nickname', 'mana', 'weaponTier'],
  });

  const combined: Omit<LeaderboardEntry, 'rank' | 'isMetaKing'>[] = [
    ...players.map(p => ({ nickname: p.nickname!, mana: p.mana, weaponTier: p.weaponTier, type: 'player' as const })),
    ...bots.map(b => ({ nickname: b.nickname, mana: b.mana, weaponTier: b.weaponTier, type: 'bot' as const })),
  ];

  combined.sort((a, b) => b.mana - a.mana);
  const top = combined.slice(0, CONSTANTS.LEADERBOARD_TOP_N);

  const entries: LeaderboardEntry[] = top.map((e, i) => ({
    ...e,
    rank: i + 1,
    isMetaKing: i === 0,
  }));

  cache = { entries, expiresAt: Date.now() + CONSTANTS.LEADERBOARD_CACHE_SECONDS * 1000 };
  return entries;
}

export function invalidateLeaderboardCache(): void {
  cache = null;
}
```

```typescript
export interface EntitlementResult {
  active: boolean;
  productIdentifier?: string;
  expiresAt?: string;
}

export async function checkEntitlement(appleUserId: string): Promise<EntitlementResult> {
  try {
    const response = await axios.get(
      `https://api.revenuecat.com/v1/subscribers/${encodeURIComponent(appleUserId)}`,
      {
        headers: {
          Authorization: `Bearer ${process.env.REVENUECAT_SECRET_KEY}`,
          'Content-Type': 'application/json',
        },
      },
    );

    const entitlements = response.data?.subscriber?.entitlements ?? {};
    const active = entitlements['metaking_access'];

    if (active && new Date(active.expires_date) > new Date()) {
      return {
        active: true,
        productIdentifier: active.product_identifier,
        expiresAt: active.expires_date,
      };
    }

    return { active: false };
  } catch (err: any) {
    if (err.response?.status === 404) return { active: false };
    throw err;
  }
}
```

```typescript
dotenv.config();
import { initDb, getEm } from '../db';
import { Bot } from '../entities/Bot';

import { rollWeapon } from '../services/weaponService';

const ADJECTIVES = ['Dark', 'Shadow', 'Iron', 'Storm', 'Frost', 'Void', 'Blood', 'Soul', 'Bone', 'Ash',
  'Silver', 'Gold', 'Ember', 'Thunder', 'Grim', 'Ancient', 'Cursed', 'Fallen', 'Hollow', 'Lost'];
const NOUNS = ['Knight', 'Mage', 'Warden', 'Reaper', 'Stalker', 'Walker', 'Seeker', 'Hunter', 'Blade',
  'Lance', 'Shield', 'Wraith', 'Specter', 'Wanderer', 'Guardian', 'Phantom', 'Oracle', 'Titan', 'Drake', 'Sage'];

function generateNickname(index: number): string {
  const adj = ADJECTIVES[index % ADJECTIVES.length];
  const noun = NOUNS[Math.floor(index / ADJECTIVES.length) % NOUNS.length];
  return `${adj}${noun}${index}`;
}

function generateMana(index: number): number {
  // Weighted distribution: 70% 100-2000, 20% 2000-10000, 9% 10000-50000, 1% 50000-100000
  const r = Math.random();
  if (r < 0.70) return Math.floor(100 + Math.random() * 1900);
  if (r < 0.90) return Math.floor(2000 + Math.random() * 8000);
  if (r < 0.99) return Math.floor(10000 + Math.random() * 40000);
  return Math.floor(50000 + Math.random() * 50000);
}

async function seedBots() {
  await initDb();
  const em = getEm();

  const existing = await em.count(Bot, {});
  if (existing >= CONSTANTS.BOT_POOL_SIZE) {
    console.log(`Already have ${existing} bots, skipping seed.`);
    process.exit(0);
  }

  const toCreate = CONSTANTS.BOT_POOL_SIZE - existing;
  console.log(`Creating ${toCreate} bots...`);

  const BATCH_SIZE = 500;
  for (let i = existing; i < CONSTANTS.BOT_POOL_SIZE; i += BATCH_SIZE) {
    const batch = [];
    for (let j = i; j < Math.min(i + BATCH_SIZE, CONSTANTS.BOT_POOL_SIZE); j++) {
      const mana = generateMana(j);
      const weapon = rollWeapon(mana);
      // Ensure ~70% start offline to populate portal pool
      const isOnline = Math.random() < 0.30;
      batch.push(em.create(Bot, {
        nickname: generateNickname(j),
        mana,
        weaponTier: weapon.tier,
        isOnline,
      }));
    }
    await em.persistAndFlush(batch);
    console.log(`  Created ${Math.min(i + BATCH_SIZE, CONSTANTS.BOT_POOL_SIZE)} / ${CONSTANTS.BOT_POOL_SIZE}`);
  }

  console.log('Bot seed complete.');
  process.exit(0);
}

seedBots().catch(err => { console.error(err); process.exit(1); });
```

```typescript
import { EntityManager } from '@mikro-orm/postgresql';
import { Bot } from '../entities/Bot';
import { Player } from '../entities/Player';
import { Message } from '../entities/Message';
import { CONSTANTS } from '../config/constants';
import { rollWeapon } from './weaponService';

const EMOJIS = ['👑', '⚔️', '💀', '🔥', '👻', '😈', '🙏', '✨', '😂', '🤝', '💎', '🌀'];

function randomEmoji(): string {
  return EMOJIS[Math.floor(Math.random() * EMOJIS.length)];
}

function driftMana(mana: number): number {
  const drift = CONSTANTS.BOT_MANA_DRIFT_PERCENT;
  const factor = 1 + (Math.random() * drift * 2 - drift);
  return Math.max(CONSTANTS.MANA_MIN, Math.round(mana * factor));
}

export async function runBotCycle(em: EntityManager): Promise<void> {
  const allBots = await em.find(Bot, {});
  const totalBots = allBots.length;
  if (totalBots === 0) return;

  // Step 1: Availability flip
  for (const bot of allBots) {
    if (Math.random() < CONSTANTS.BOT_ONLINE_FLIP_PROBABILITY) {
      bot.isOnline = !bot.isOnline;
    }
  }

  // Enforce offline floor
  const offlineBots = allBots.filter(b => !b.isOnline);
  const minOffline = Math.ceil(totalBots * CONSTANTS.BOT_OFFLINE_FLOOR_PERCENT);
  if (offlineBots.length < minOffline) {
    const onlineBots = allBots.filter(b => b.isOnline);
    const toFlip = minOffline - offlineBots.length;
    for (let i = 0; i < toFlip && i < onlineBots.length; i++) {
      onlineBots[i].isOnline = false;
    }
  }

  // Step 2: Mana drift
  for (const bot of allBots) {
    bot.mana = driftMana(bot.mana);
    const weapon = rollWeapon(bot.mana);
    bot.weaponTier = weapon.tier;
    bot.lastCronAt = new Date();
  }

  await em.flush();

  // Step 3: Bot visits (10% of bots, each visits 2 targets)
  const visitorCount = Math.ceil(totalBots * CONSTANTS.BOT_VISITOR_PERCENT);
  const shuffled = [...allBots].sort(() => Math.random() - 0.5);
  const visitors = shuffled.slice(0, visitorCount);

  // Build offline pool: offline bots + offline real players
  const offlinePlayerTargets = await em.find(Player, { isOnline: false });
  const currentOfflineBots = allBots.filter(b => !b.isOnline);

  type Target = { type: 'player'; entity: Player } | { type: 'bot'; entity: Bot };
  const pool: Target[] = [
    ...offlinePlayerTargets.map(p => ({ type: 'player' as const, entity: p })),
    ...currentOfflineBots.map(b => ({ type: 'bot' as const, entity: b })),
  ];

  if (pool.length === 0) {
    await em.flush();
    return;
  }

  for (const visitor of visitors) {
    for (let v = 0; v < CONSTANTS.BOT_VISITS_PER_ACTIVE_BOT; v++) {
      const target = pool[Math.floor(Math.random() * pool.length)];
      if (target.type === 'bot' && target.entity.id === visitor.id) continue;

      target.entity.mana = Math.max(CONSTANTS.MANA_MIN, target.entity.mana - CONSTANTS.PORTAL_MANA_STEAL);
      visitor.mana += CONSTANTS.PORTAL_MANA_GAIN;

      if (target.type === 'player') {
        const msg = em.create(Message, {
          fromId: visitor.id,
          fromNickname: visitor.nickname,
          toPlayerId: target.entity.id,
          emoji: randomEmoji(),
          manaStolen: CONSTANTS.PORTAL_MANA_STEAL,
        });
        em.persist(msg);
      }
    }
  }

  // Re-update weapon tiers for visitors who gained mana
  for (const visitor of visitors) {
    const weapon = rollWeapon(visitor.mana);
    visitor.weaponTier = weapon.tier;
  }

  await em.flush();
}
```
